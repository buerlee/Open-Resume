% Section header
\renewcommand{\section}[2] {
    \pagebreak[2]\vspace{0.5\baselineskip}
    \phantomsection\addcontentsline{toc}{section}{#1}%
	\textbf{\large #1}
    \vspace{3pt}\noindent\hrule
}

% Three-column entry
\newcommand{\entry}[3] {
    \par\vspace{1pt}\noindent
    \makebox[0pt][l]{#1}%
    \makebox[\textwidth][c]{#2}%
    \makebox[0pt][r]{#3}
    \par\vspace{1pt}
}

% Bold-face version of entry
\newcommand{\entrybf}[3] {
    \entry{\textbf{#1}}{\textbf{#2}}{\textbf{#3}}
}

% Version of entry for describing employer and position
\newcommand{\jobentry}[3] {
    \entry{\textbf{#1} #2}{}{\textbf{#3}}
}

% Custom square bullet for lists
\newcommand{\squarebullet}{\textcolor{black}{\raisebox{.4ex}{\rule{.6ex}{.6ex}}}}

% List for within entries
\newenvironment{entrylist}[1][\enskip\squarebullet]
    {\begin{compactitem} [#1]}
    {\end{compactitem}\vspace{0.3\baselineskip}}

% List offset and aligned with description text
\newenvironment{descriptionlist}[1]
    {\vspace{-1.1\baselineskip}
        \begin{description}[labelwidth=\widthof{#1}+0.5em,leftmargin=\dimexpr\labelwidth+\labelsep\relax]
        \item[#1]
        \begin{compactitem}[\squarebullet]}
    {\end{compactitem}
        \end{description}
        \vspace{-.5\baselineskip}}

% Version of descriptionlist intended for honors
\newenvironment{entryhonorlist}
    {\begin{descriptionlist}{\enskip\textmd{Honors:}}}
    {\end{descriptionlist}}

% Height of line marking sections
\newcommand{\hlineheight}{0.1pt}
